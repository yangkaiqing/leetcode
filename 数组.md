### 1、找出数组中出现超过一半的元素

```python
def majorityElement(self, nums: List[int]) -> int:
        if not nums:
            return -1
        ticket = 1
        people = nums[0]
        for x in range(1,len(nums)):
            if ticket == 0:
                people = nums[x]
            if people != nums[x]:
                ticket -= 1
            else:
                ticket += 1
        if ticket > 0:
            count = 0
            for elem in nums:
                if elem == people:
                    count += 1
                    if count > len(nums)//2:
                        return people
            return -1
```
