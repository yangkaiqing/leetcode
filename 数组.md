## 数组

### 1、找出数组中出现超过一半的元素

```python
def majorityElement(self, nums: List[int]) -> int:
        if not nums:
            return -1
        ticket = 1
        people = nums[0]
        for x in range(1,len(nums)):
            if ticket == 0:
                people = nums[x]
            if people != nums[x]:
                ticket -= 1
            else:
                ticket += 1
        if ticket > 0:
            count = 0
            for elem in nums:
                if elem == people:
                    count += 1
                    if count > len(nums)//2:
                        return people
            return -1
```

2、获取第K个满足某个条件的数，比如丑数，比如公因式只有3，5，7的

```python
def getKthMagicNumber(self, k: int) -> int:
        dp = (k) * [0]
        dp[0] = 1
        x = 0
        y = 0
        z = 0
        for i in range(1,k):
            dp[i] = min(3*dp[x], 5*dp[y], 7*dp[z])
            if dp[i] == 3*dp[x]:
                x += 1
            if dp[i] == 5*dp[y]:
                y += 1
            if dp[i] == 7*dp[z]:
                z += 1
        return dp[k-1]
```
